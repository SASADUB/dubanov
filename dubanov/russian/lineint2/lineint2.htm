function F=F3(X,a,b,c,D,E)

F(:,1)=a(1)*X(:,1)+b(1)*X(:,2)+...

       c(1)*X(:,1).*X(:,2)+...

       D(:,1).*X(:,3)+E(:,1);

F(:,2)=a(2)*X(:,1)+b(2)*X(:,2)+...

       c(2)*X(:,1).*X(:,2)+...

       D(:,2).*X(:,3)+E(:,2);

F(:,3)=a(3)*X(:,1)+b(3)*X(:,2)+...

       c(3)*X(:,1).*X(:,2)+...

       D(:,3).*X(:,3)+E(:,3);
