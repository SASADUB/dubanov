Преобразование

            Рассмотрим ситуацию, когда преобразованию подвергается не сегмент параметрической кривой , а сегмент параметрической поверхности . В принципе, коренных различий нет. Еще раз отметим, что все зависит от способа хранения вектора.

function Segment4()

clc, clear

alpha=pi/6; C=[1.5;1;0];

[u,v]=meshgrid(0:2*pi/100:pi/4,0:pi/100:pi/2);

x=cos(u).*sin(v); y=sin(u).*sin(v); z=cos(v);

mesh(x,y,z), axis([-2 2 -2 2 -2 2]);

R(:,:,1)=x-C(1);

R(:,:,2)=y-C(2);

R(:,:,3)=z-C(3);

E1=zeros(size(R));  

E2=zeros(size(R));

E3=zeros(size(R));

E1(:,:,1)=cos(alpha); 

E1(:,:,2)=sin(alpha);

E1(:,:,3)=0;

E2(:,:,1)=-sin(alpha);

E2(:,:,2)=cos(alpha);

E2(:,:,3)=0;

E3(:,:,1)=0;          

E3(:,:,2)=0;         

E3(:,:,3)=1;

x1=dot(R,E1,3);

y1=dot(R,E2,3);

z1=dot(R,E3,3);

hold on, mesh(x1,y1,z1), axis([-2 2 -2 2 -2 2]), grid on;

 

R1(:,:,1)=x1;

R1(:,:,2)=y1;

R1(:,:,3)=z1;

 

K(:,:,1)=dot(cross(E1,E2,3),E3,3);

K(:,:,2)=dot(cross(E1,E2,3),E3,3);

K(:,:,3)=dot(cross(E1,E2,3),E3,3);

 

Ev1=cross(E2,E3,3)./K;

Ev2=cross(E3,E1,3)./K;

Ev3=cross(E1,E2,3)./K;

 

H1=zeros(size(R)); H1(:,:,1)=1;

H2=zeros(size(R)); H2(:,:,2)=1;

H3=zeros(size(R)); H3(:,:,3)=1;

 

Hp1(:,:,1)=dot(H1,Ev1,3);

Hp1(:,:,2)=dot(H1,Ev2,3);

Hp1(:,:,3)=dot(H1,Ev3,3);

Hp2(:,:,1)=dot(H2,Ev1,3);

Hp2(:,:,2)=dot(H2,Ev2,3);

Hp2(:,:,3)=dot(H2,Ev3,3);

Hp3(:,:,1)=dot(H3,Ev1,3);

Hp3(:,:,2)=dot(H3,Ev2,3);

Hp3(:,:,3)=dot(H3,Ev3,3);

 

x2=dot(R1,Hp1,3)+C(1);

y2=dot(R1,Hp2,3)+C(2);

z2=dot(R1,Hp3,3)+C(3);

hold on, mesh(x2,y2,z2)
